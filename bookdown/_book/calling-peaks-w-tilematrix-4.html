#Call Reproducible Peaks w/ TileMatrix (~2 minutes)
projHemeTmp <- addReproduciblePeakSet(
    ArchRProj = projHeme4, 
    groupBy = "Clusters",
    peakMethod = "Tiles",
    method = "p"
)
